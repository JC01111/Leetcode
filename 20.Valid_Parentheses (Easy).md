### My Solution:
```
class Solution:
    def isValid(self, s: str) -> bool:
        stack = []  # stack is LIFO
        for i in s:                
            if i == '(' or i == '[' or i == '{':
                stack.append(i)
            else:
                # Case when s = ["]"]
                if not stack:
                    return False
                opening = stack.pop()
                if i == ')' and opening != '(':
                    return False
                elif i == ']' and opening != '[':
                    return False
                elif i == '}' and opening != '{':
                    return False
        # If any opening left means there is no matching
        if stack:
            return False
        return True
```
